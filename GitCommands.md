Setup:
- git config --global user.name “[firstname lastname]”
- git config --global user.email “[valid-email]”
- git config --global color.ui auto
- git config --global pull.rebase true


Basics:
- git status
- git add .
- git commit -m 
- git push
- git pull

Branching:
- git branch
- git checkout -b hchen/working
